improve_civilian_industrial_projects = {
	if = {
		limit = { NOT = { has_idea = civilian_industry_project } }
		add_ideas = civilian_industry_project
	}
	else_if = {
		limit = { has_idea = civilian_industry_project }
		swap_ideas = {
			remove_idea = civilian_industry_project
			add_idea = civilian_industry_project2
		}
	}
	else_if = {
		limit = { has_idea = civilian_industry_project2 }
		swap_ideas = {
			remove_idea = civilian_industry_project2
			add_idea = civilian_industry_project3
		}
	}
}
